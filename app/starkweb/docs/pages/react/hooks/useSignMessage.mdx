# useSignMessage

Hook for signing messages.

## Import

```ts
import { useSignMessage } from "sn-wolf"
```

## Usage

```tsx [index.tsx]
import { useSignMessage } from "sn-wolf"

function App() {
  const { signMessage } = useSignMessage()

  return (
    <button onClick={() => signMessage({ message: 'hello world' })}>
      Sign message
    </button>
  )
}
```
