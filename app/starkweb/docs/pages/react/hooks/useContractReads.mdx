# useContractReads

Hook for calling multiple read methods on a contract.

## Import

```ts
import { useContractReads } from "sn-wolf"
```

## Usage

```tsx [index.tsx]
import { useContractReads } from "sn-wolf"

const gotchiContract = {
  address: "0x072589b1e65b5b22b746c8701aae3ecc34af6e64a378091b47188e2a52414dba",
  abi: gotchiABI,
} as const
const mlootContract = {
  address: "0x06707675cd7dd9256667eca8284e46f4546711ee0054bc2dd02f0ce572056cf4",
  abi: mlootABI,
} as const

function App() {
  const result = useContractReads({
    contracts: [
      {
        ...gotchiContract,
        functionName: "getAlive",
      },
      {
        ...gotchiContract,
        functionName: "getBoredom",
      },
      {
        ...mlootContract,
        functionName: "getChest",
        args: [69],
      },
      {
        ...mlootContract,
        functionName: "getWaist",
        args: [69],
      },
    ],
  })
}
```

### allowFailure

`boolean`

Whether or not the Hook should throw if a call reverts. If set to `true` (default), and a call reverts, then `useContractReads` will fail silently and its error will be logged in the results array. Defaults to `true`.

```tsx [index.tsx]
import { useContractReads } from "sn-wolf"

function App() {
  const result = useContractReads({
    allowFailure: false, // [!code focus]
    contracts: [
      {
        address: "0x06707675cd7dd9256667eca8284e46f4546711ee0054bc2dd02f0ce572056cf4",
        abi: mlootABI,
        functionName: "getChest",
        args: [69]
      },
      // ...
    ],
  })
}
```
